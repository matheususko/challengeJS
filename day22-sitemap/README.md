```xml (sitemap.xml)
<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
  <url>
    <loc>http://example.com/foo.html</loc>
    <lastmod>2018-06-04</lastmod>
  </url>
</urlset>

```
